spherical	Spherical	q = vec3(length(q), acos(q.z / length(q)), atan(q.y, q.x))

cylindricaly	Cylindrical Y	q = vec3(atan(q.x, q.z), q.y, length(q.xz))
cylindricalz	Cylindrical Z	q = vec3(length(q.xy), atan(q.y, q.x), q.z)
logcylindricalx	Log-Cylindrical X	q = vec3(q.x, log(length(q.yz)), atan(q.z, q.y))
logcylindricaly	Log-Cylindrical Y	q = vec3(atan(q.x, q.z), q.y, log(length(q.xz)))
logcylindricalz	Log-Cylindrical Z	q = vec3(log(length(q.xy)), atan(q.y, q.x), q.z)
